#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

void Problem7(double *a, double *b, double *c, double *u, double *f, int n, double *bs, double *fs)
  {
 bs[1] = b[1];
 fs[1] = f[1];

 double metadata;

 for (int i = 2; i < n+1; i++)

  {
      metadata = a[i-1]/bs[i-1];
      bs[i] = b[i] - c[i-1]*metadata;
      fs[i] = f[i] - fs[i-1]*metadata;
  }

  u[n] = fs[n]/bs[n];
  for (int i = n-1; i>0; i--)
  {
  u[i] = (fs[i] - c[i]*u[i+1])/bs[i];
  }
  }
